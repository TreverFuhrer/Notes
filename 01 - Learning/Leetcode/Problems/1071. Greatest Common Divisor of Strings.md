# Code written by Trever Fuhrer
class Solution:
	def gcdOfStrings(self, str1: str, str2: str) -> str:
		if (str1 + str2) != (str2 + str1):
			return ""

		k, j = sorted([len(str1), len(str2)])
		while(j % k):
			# Easy python way to not use temp
			# j = k
			# k = j%k
			(j, k) = k, j % k

		return str1[:k]
