```python
# Code written by Trever Fuhrer
class Solution:
	def isSubsequence(self, s: str, t: str) -> bool:
		n = len(s)
		if n == 0:
			return True

		p1 = 0
		for p2 in t:
			if s[p1] == p2:
				p1 += 1
			if p1 == n:
				return True
		return False
```
